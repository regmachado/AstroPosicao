\documentclass[border=1mm]{standalone}

\usepackage{preambletikz}

\begin{document}

% 3D projection
\tdplotsetmaincoords{70}{100}

\begin{tikzpicture}[tdplot_main_coords, scale=3.0]
 
\def\gama{20}
\tdplotsetrotatedcoords{-\gama}{0}{0};

% south celestial pole
\draw[-, thick, tdplot_rotated_coords] (0,0,0) -- (0,0,-1.06) node[below] {Pólo Sul Celeste};

% equator
\draw[fill, tdplot_rotated_coords, purpleequator, opacity=0.75] (0,0,0) circle (1);

% celestial sphere
\shade[ball color=bluesky, opacity=0.5] (0,0,0) circle (1cm);

% equator
\draw[tdplot_rotated_coords, decoration={text along path, text={Equador Celeste}, text align={center}}, decorate] (0.8,0.4) to [bend right=14] (-0.4,1);

% north celestial pole
\draw[->, thick, tdplot_rotated_coords] (0,0,0) -- (0,0,1.2) node[above] {Pólo Norte Celeste};

% vernal point
\draw[black, fill=black, tdplot_rotated_coords] (1,0,0) circle (0.4pt) node[below]{\Aries};
 
\end{tikzpicture}
 
\end{document}
